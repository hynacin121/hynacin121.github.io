# __Rebalancing__

+ Portfolio Rebalancing refers to re-weighting the portfolio

## __Buy and Hold__

+ Investment that is not re-allocated

## __Rebalancing__

+ Re-allocating portfolio at certain intervals
+ Example:
    + Portfolio : 25% in Apple, 75% in Twitter
    + Rebalance at the end of each month
    + portfolio is allocated to 25%, 75% at the end of each month
+ Rebalancing doesn't have to re-allocate to original weight
+ Rebalancing doesn't always have to occur at pre-determined interval.
+ It is possible to set a rebalancing period and re-allocate at the end of each rebalancing period

# __Backtest__

+ Use Historical returns to check the performance of a portfolio model.

+ Rebalance period : Re-allocate period
+ Lookback period : Recent historical period used for estimating inputs of the model

class Portfolio_6040():

        self.name = name
        
        # list of class asset elements

        dates = [each.get_date() for each in self.assets]
        lendates = [len(each) for each in dates]

        # 이후 거래일

            total_balance_tmp = 0

                # 다음 리밸런싱 날짜 업데이트
                next_rebalancing.append(next_rebalancing[i-1] + relativedelta(months=self.rebalancing_interval))
                
                
                weight = self.ratios
                    
                for j in range(len(self.assets)):
                    # i시점의 j자산에 대한 밸런스 계산
                    balance = total_balance[i-1] * weight[j] * (1 + self.assets[j].get_change()[i])
                    # 자산별 밸런스 리스트에 추가
                    balances[j].append(balance)
                    # i 시점의 전체 밸런스 계산
                    total_balance_tmp += balances[j][i]
            
            # 리밸런싱 하지 않는 날
            else:
                # 이전 리밸런싱 날짜랑 동일하게
                next_rebalancing.append(next_rebalancing[i-1])
                # Asset 번호
                for j in range(len(self.assets)):
                    try:
                        balances[j].append(balances[j][i-1] * (1 + self.assets[j].get_change()[i]))
                    except:
                        balances[j].append(balances[j][i-1])
                    total_balance_tmp += balances[j][i]
            
            # 밸런스 업데이트
            total_balance.append(total_balance_tmp)

           

        df = pd.DataFrame()
        df['Date'] = self.date
        df.set_index('Date', inplace=True)
        df['Total'] = total_balance

        for i in range(len(self.assets)):
            df[self.assets[i].get_name()] = balances[i]
        
        print('Portfolio Backtest Complete')
        self.backtest_df = df
        
        # 밸런스 DataFrame 반환
        return df
